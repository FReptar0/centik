    **Data fetching pattern:**
    This is a Server Component. Use the async pattern from Phase 6 (Next.js 16 async searchParams).

    ```typescript
    interface PageProps {
      searchParams: Promise<{ month?: string; year?: string }>
    }

    export default async function HomePage({ searchParams }: PageProps) {
      const params = await searchParams
      
      // Resolve period (respect URL params for historical periods)
      // If month/year params exist, find that period; otherwise use current
      let period: Period
      if (params.month && params.year) {
        period = await getPeriodForDate(`${params.year}-${String(params.month).padStart(2, '0')}-01`)
      } else {
        period = await getCurrentPeriod()
      }

      // Parallel data fetch -- all 5 queries at once
      const [kpis, categoryExpenses, budgetVsSpent, trend, recentTransactions] = await Promise.all([
        getDashboardKPIs(period.id),
        getCategoryExpenses(period.id),
        getBudgetVsSpent(period.id),
        getMonthlyTrend(),
        getRecentTransactions(period.id),
      ])

      return (
        <div className="space-y-6">
          <PageHeader
            title="Inicio"
            periodSelector={<PeriodSelector isClosed={period.isClosed} />}
            closedBanner={period.isClosed}
          />

          <KPIGrid kpis={kpis} />

          <div className="grid grid-cols-1 lg:grid-cols-2 gap-4">
            <BudgetBarChart data={budgetVsSpent} />
            <ExpenseDonutChart data={categoryExpenses} />
          </div>

          <div className="grid grid-cols-1 lg:grid-cols-2 gap-4">
            <TrendAreaChart data={trend} />
            <RecentTransactions transactions={recentTransactions} />
          </div>
        </div>
      )
    }
    ```

    Key requirements:
    - Import serializeBigInts from @/lib/serialize -- the dashboard query functions should already return serialized data (strings), but if any raw Prisma results sneak through, serialize at the boundary
    - The page must remain a Server Component (NO "use client")
    - All chart components and KPIGrid are Client Components (they have "use client")
    - PeriodSelector receives isClosed from the resolved period
    - Pass closedBanner to PageHeader when viewing a closed period
  </action>
